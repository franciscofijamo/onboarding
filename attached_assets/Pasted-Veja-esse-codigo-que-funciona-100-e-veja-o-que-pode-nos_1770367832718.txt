import { createPublicKey, publicEncrypt, constants } from "crypto";

function toPemFromRawDerBase64(rawDerBase64: string): string {
  // Your PHP does: base64_decode(raw_public_key_base64) then base64_encode again -> same payload.
  // We'll decode then re-encode to ensure consistent formatting.
  const der = Buffer.from(rawDerBase64, "base64");
  const b64 = der.toString("base64");
  const lines = b64.match(/.{1,64}/g) ?? [];
  return `-----BEGIN PUBLIC KEY-----\n${lines.join("\n")}\n-----END PUBLIC KEY-----`;
}

function encryptApiKeyToBearer(apiKey: string, publicKeyPem: string): string {
  const keyObj = createPublicKey(publicKeyPem);
  const encrypted = publicEncrypt(
    {
      key: keyObj,
      padding: constants.RSA_PKCS1_PADDING, // matches PHP openssl_public_encrypt default
    },
    Buffer.from(apiKey, "utf8")
  );
  return encrypted.toString("base64");
}

function thirdPartyReference(onId: string, dateObj = new Date()): string {
  // PHP: 'ON' .$on_id.'Time'. date('H\hi')  -> e.g. ON1Time14h05
  const hh = String(dateObj.getHours()).padStart(2, "0");
  const mm = String(dateObj.getMinutes()).padStart(2, "0");
  return `ON${onId}Time${hh}h${mm}`;
}

async function logToFile(sender: string, message: string): Promise<void> {
  const now = new Date();
  const dd = String(now.getDate()).padStart(2, "0");
  const MM = String(now.getMonth() + 1).padStart(2, "0");
  const yy = String(now.getFullYear()).slice(-2);
  const HH = String(now.getHours()).padStart(2, "0");
  const mm = String(now.getMinutes()).padStart(2, "0");
  const ss = String(now.getSeconds()).padStart(2, "0");

  const fileName = `mobile_wallet_${dd}-${MM}-${yy}.txt`;
  const line = `${sender} ${message} ${dd}/${MM}/${yy} ${HH}:${mm}:${ss}\n`;
  await appendFile(fileName, line, { encoding: "utf8" });
}
